Module qscope.examples.role_system_example

Example demonstrating the QScope role system architecture.

This example shows how the role system connects protocols, interfaces, roles, and devices to provide a flexible hardware abstraction layer. It includes:

  1. Creating a new protocol
  2. Creating a new interface
  3. Creating a new role
  4. Implementing a device for the role
  5. Using the device through the role interface

The role system allows for hardware abstraction, making it possible to swap device implementations without changing measurement code.

def main():
    """Demonstrate the role system in action."""
    # Import system classes
    from qscope.system import System
    from qscope.system.base_config import SystemConfig
    
    # Create a simple system config
    sys_config = SystemConfig(
        system_name="example_system",
        system_type=System  # Pass the System class itself
    )
    
    # Create a system with the config
    system = System(sys_config)
    
    # Create device instance
    motor = MockMotorController()
    
    # Add device with role
    system.add_device_with_role(motor, MOTOR_CONTROLLER)
    
    # Get device by role (returns interface)
    motor_controller = system.get_device_by_role(MOTOR_CONTROLLER)
    
    # Use interface methods
    motor_controller.home()
    motor_controller.move_to_position(10.5)
    current_position = motor_controller.get_position()
    print(f"Current position: {current_position} mm")
    
    # This is the key benefit: measurements work with roles, not specific devices
    # We could replace MockMotorController with a real hardware implementation
    # without changing any of the code that uses it

class MockMotorController(Device):
    """Mock implementation of a motor controller.
    
    This class provides a simulated motor controller for testing
    and demonstration purposes. It implements the MotorControllerProtocol
    and can be used with the MotorController role.
    """
    
    def __init__(self, **config_kwargs):
        """Initialize the mock motor controller.
        
        Parameters
        ----------
        **config_kwargs
            Configuration parameters
        """
        super().__init__(**config_kwargs)
        self._connected = False
        self._position = 0.0
    
    def open(self) -> tuple[bool, str]:
        """Connect to the device.
        
        Returns
        -------
        tuple[bool, str]
            Success status and message
        """
        self._connected = True
        return True, "Connected to mock motor controller"
    
    def close(self):
        """Disconnect from the device."""
        self._connected = False
    
    def is_connected(self) -> bool:
        """Check if the device is connected.
        
        Returns
        -------
        bool
            True if connected, False otherwise
        """
        return self._connected
    
    # Implement MotorControllerProtocol methods
    def move_to_position(self, position: float) -> None:
        """Move to the specified position.
        
        Parameters
        ----------
        position : float
            Target position in mm
        """
        print(f"Moving to position {position} mm")
        self._position = position
    
    def get_position(self) -> float:
        """Get the current position.
        
        Returns
        -------
        float
            Current position in mm
        """
        return self._position
    
    def home(self) -> None:
        """Move to the home position."""
        print("Homing motor")
        self._position = 0.0

class MotorControllerInterface(RoleInterface):
    """Interface for motor controller devices.
    
    This interface provides methods for controlling motorized stages
    or positioners. It wraps a device implementing MotorControllerProtocol
    and provides a clean, type-safe API for motor control operations.
    """
    
    def __init__(self, device: MotorControllerProtocol):
        """Initialize the motor controller interface.
        
        Parameters
        ----------
        device : MotorControllerProtocol
            Device implementing the MotorControllerProtocol
        """
        super().__init__(device)
    
    def move_to_position(self, position: float) -> None:
        """Move motor to specified position.
        
        Parameters
        ----------
        position : float
            Target position in mm
        """
        return self._device.move_to_position(position)
    
    def get_position(self) -> float:
        """Get current motor position.
        
        Returns
        -------
        float
            Current position in mm
        """
        return self._device.get_position()
    
    def home(self) -> None:
        """Move motor to home position."""
        return self._device.home()
